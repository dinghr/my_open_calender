## 第二部分：静态登录码系统

### 2.1 需求说明

**背景：** 微信小程序认证需要企业资质（费用高），我们使用静态登录码方案。

**方案：**
```
1. 系统预先生成一批登录码
2. 家长使用登录码登录
3. 首次登录自动识别为管理员
4. 管理员可以绑定孩子
5. 孩子使用登录码登录（绑定后）
```

**登录码格式：**
```
FAMILY-XXXX-XXXX
例如：FAMILY-A1B2-C3D4
```

### 2.4 生成登录码脚本

**文件：** `scripts/generate_login_codes.py`

```python
#!/usr/bin/env python3
"""生成静态登录码"""
import sys
import os
import random
import string

sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..', 'backend'))

from models import Database, LoginCode, LoginCodeStatus
from datetime import datetime, timedelta
from uuid import uuid4

def generate_code():
    """生成登录码：FAMILY-XXXX-XXXX"""
    chars = string.ascii_uppercase + string.digits
    part1 = ''.join(random.choices(chars, k=4))
    part2 = ''.join(random.choices(chars, k=4))
    return f"FAMILY-{part1}-{part2}"

def generate_login_codes(count=100, expire_days=365, max_uses=None):
    """生成一批登录码"""
    db = Database("data/prod.db")
    session = db.get_session()
    
    codes = []
    for _ in range(count):
        code = generate_code()
        
        # 检查是否重复
        while session.query(LoginCode).filter(LoginCode.code == code).first():
            code = generate_code()
        
        login_code = LoginCode(
            id=str(uuid4()),
            code=code,
            status=LoginCodeStatus.ACTIVE,
            max_uses=max_uses,
            expires_at=datetime.utcnow() + timedelta(days=expire_days) if expire_days else None,
            note="生产环境登录码"
        )
        
        session.add(login_code)
        codes.append(code)
    
    session.commit()
    session.close()
    
    return codes

if __name__ == "__main__":
    count = int(sys.argv[1]) if len(sys.argv) > 1 else 100
    
    print(f"生成 {count} 个登录码...")
    codes = generate_login_codes(count)
    
    # 输出到文件
    output_file = f"login_codes_{datetime.now().strftime('%Y%m%d_%H%M%S')}.txt"
    with open(output_file, 'w') as f:
        for code in codes:
            f.write(f"{code}\n")
    
    print(f"✅ 生成成功！")
    print(f"📁 文件：{output_file}")
    print(f"\n示例登录码：")
    for code in codes[:5]:
        print(f"  {code}")
```

## 第三部分：管理员与孩子绑定

### 3.1 首次登录识别管理员

**流程：**
```
1. 用户使用登录码首次登录
2. 系统检查该登录码是否绑定用户
3. 未绑定 → 创建新用户，标记为管理员
4. 绑定登录码与用户
5. 返回 JWT token
```

**API 实现：**

```python
@router.post("/auth/login")
def login(
    data: LoginRequest,
    session: Session = Depends(get_db)
):
    """使用登录码登录"""
    # 1. 验证登录码
    login_code = session.query(LoginCode).filter(
        LoginCode.code == data.code
    ).first()
    
    if not login_code:
        raise HTTPException(400, "登录码无效")
    
    if login_code.status != LoginCodeStatus.ACTIVE:
        raise HTTPException(400, "登录码已失效")
    
    # 2. 检查是否已绑定用户
    user = session.query(User).filter(
        User.login_code_id == login_code.id
    ).first()
    
    # 3. 未绑定则创建新用户（管理员）
    if not user:
        user = User(
            id=str(uuid4()),
            login_code_id=login_code.id,
            nickname=f"用户{login_code.code[-8:]}",  # 临时昵称
            role=UserRole.ADMIN  # 首次登录自动为管理员
        )
        session.add(user)
        
        # 更新登录码使用次数
        login_code.current_uses += 1
    
    # 4. 生成 JWT token
    token = create_jwt_token(user.id)
    
    return {
        "token": token,
        "user": {
            "id": user.id,
            "nickname": user.nickname,
            "role": user.role.value,
            "is_first_login": user.role == UserRole.ADMIN
        }
    }
```

### 3.2 管理员绑定孩子

**流程：**
```
1. 管理员登录
2. 创建孩子账号
3. 创建绑定关系（自动审批）
4. 生成孩子的登录码
```

**API 实现：**

```python
@router.post("/children")
def create_child(
    data: ChildCreateRequest,
    user: User = Depends(get_current_user),  # JWT 认证
    session: Session = Depends(get_db)
):
    """管理员创建并绑定孩子"""
    # 1. 检查权限
    if user.role != UserRole.ADMIN:
        raise HTTPException(403, "只有管理员可以绑定孩子")
    
    # 2. 创建孩子账号
    child = Student(
        id=str(uuid4()),
        name=data.name,
        grade=data.grade,
        birthday=data.birthday
    )
    session.add(child)
    
    # 3. 创建绑定关系（自动审批）
    binding = UserStudentBinding(
        id=str(uuid4()),
        user_id=user.id,
        student_id=child.id,
        relation=data.relation,  # father, mother, etc.
        status=ApprovalStatus.APPROVED,
        approved_at=datetime.utcnow()
    )
    session.add(binding)
    
    # 4. 生成孩子的登录码
    login_code = LoginCode(
        id=str(uuid4()),
        code=generate_code(),
        status=LoginCodeStatus.ACTIVE,
        max_uses=None,  # 无限次使用
        note=f"孩子 {child.name} 的登录码"
    )
    session.add(login_code)
    
    session.commit()
    
    return {
        "child": {
            "id": child.id,
            "name": child.name,
            "grade": child.grade
        },
        "login_code": login_code.code,
        "message": "孩子账号创建成功，请使用此登录码登录"
    }
```

### 3.3 多教育者支持

**场景：** 爷爷、奶奶、爸爸、妈妈都可以绑定同一个孩子

**流程：**
```
1. 孩子已有登录码
2. 其他家长使用自己的登录码登录
3. 输入孩子的登录码（或邀请码）
4. 创建绑定关系（需要管理员审批或自动审批）
```

**API 实现：**

```python
@router.post("/bindings/request")
def request_binding(
    data: BindingRequest,
    user: User = Depends(get_current_user),
    session: Session = Depends(get_db)
):
    """申请绑定孩子"""
    # 1. 验证孩子登录码
    child_code = session.query(LoginCode).filter(
        LoginCode.code == data.child_code
    ).first()
    
    if not child_code:
        raise HTTPException(400, "孩子登录码无效")
    
    # 2. 查找孩子
    child = session.query(Student).filter(
        Student.id == child_code.bound_student_id  # 假设有这个字段
    ).first()
    
    # 3. 创建绑定申请
    binding = UserStudentBinding(
        id=str(uuid4()),
        user_id=user.id,
        student_id=child.id,
        relation=data.relation,
        status=ApprovalStatus.PENDING  # 待审批
    )
    session.add(binding)
    session.commit()
    
    return {
        "message": "绑定申请已提交，等待管理员审批"
    }
```

#!/usr/bin/env python3
"""初始化知识点库（示例：三年级数学上学期）"""
import sys
import os

sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..', 'backend'))

from models import Database, KnowledgePoint, Subject
from uuid import uuid4

def init_knowledge_points():
    db = Database("data/prod.db")
    session = db.get_session()
    
    # 检查是否已有数据
    if session.query(KnowledgePoint).count() > 0:
        print("⚠️ 知识点库已存在数据")
        return
    
    # 三年级数学上学期知识点
    knowledge_points = [
        {
            "unit": "第一单元：时、分、秒",
            "points": [
                {"name": "认识时间单位", "learning_points": ["1 时=60 分", "1 分=60 秒"]},
                {"name": "时间计算", "learning_points": ["时间加减法", "经过时间计算"]},
            ]
        },
        {
            "unit": "第二单元：万以内的加减法",
            "points": [
                {"name": "两位数加减法", "learning_points": ["口算", "笔算"]},
                {"name": "三位数加减法", "learning_points": ["进位加法", "退位减法"]},
            ]
        },
        # ... 更多单元
    ]
    
    count = 0
    for unit_data in knowledge_points:
        # 创建单元知识点（父级）
        unit_point = KnowledgePoint(
            id=str(uuid4()),
            subject=Subject.MATH,
            grade=3,
            semester="fall",
            unit=unit_data["unit"],
            name=unit_data["unit"].split("：")[1] if "：" in unit_data["unit"] else unit_data["unit"],
            learning_points=[],
            question_count=0
        )
        session.add(unit_point)
        session.flush()
        
        # 创建子知识点
        for point_data in unit_data["points"]:
            point = KnowledgePoint(
                id=str(uuid4()),
                subject=Subject.MATH,
                grade=3,
                semester="fall",
                unit=unit_data["unit"],
                parent_id=unit_point.id,
                name=point_data["name"],
                learning_points=point_data["learning_points"],
                question_count=0
            )
            session.add(point)
            count += 1
    
    session.commit()
    session.close()
    
    print(f"✅ 初始化完成！共 {count} 个知识点")

if __name__ == "__main__":
    init_knowledge_points()

@router.get("/students/{student_id}/progress")
def get_student_progress(
    student_id: str,
    date_from: str = None,
    date_to: str = None,
    session: Session = Depends(get_db)
):
    """获取学生学习进度"""
    # 1. 查询学习记录
    query = session.query(LearningRecord).filter(
        LearningRecord.student_id == student_id
    )
    
    if date_from:
        query = query.filter(LearningRecord.date >= date_from)
    if date_to:
        query = query.filter(LearningRecord.date <= date_to)
    
    records = query.all()
    
    # 2. 统计
    total = len(records)
    completed = sum(1 for r in records if r.status == RecordStatus.COMPLETED)
    pending = sum(1 for r in records if r.status == RecordStatus.PENDING)
    
    # 3. 按科目统计
    by_subject = {}
    for r in records:
        subject = r.subject.value if r.subject else "unknown"
        if subject not in by_subject:
            by_subject[subject] = {"total": 0, "completed": 0}
        by_subject[subject]["total"] += 1
        if r.status == RecordStatus.COMPLETED:
            by_subject[subject]["completed"] += 1
    
    # 4. 知识点掌握情况
    knowledge_points_mastered = []  # TODO: 实现
    
    return {
        "student_id": student_id,
        "summary": {
            "total": total,
            "completed": completed,
            "pending": pending,
            "completion_rate": completed / total if total > 0 else 0
        },
        "by_subject": by_subject,
        "knowledge_points": knowledge_points_mastered
